-- PLAN ONLY. NOT A CONFIGURED MIGRATION. DO NOT EXECUTE AS-IS.
--
-- This is the reviewed transaction body for a future parameterized Node
-- command. The runner binds $1 = initial owner UUID and $2 = reviewed writer
-- safety artifact. It must also compare the dry-run manifest before opening
-- this transaction. No literal owner or legacy value belongs in this file.

BEGIN;

SELECT set_config('varda.initial_owner_id', $1::text, true);
SELECT set_config('varda.writer_safety', $2::text, true);
SELECT pg_advisory_xact_lock(
  hashtextextended('varda:initial-owner-backfill:v1', 0)
);

CREATE TEMP TABLE _varda_owner_tables (
  table_name text PRIMARY KEY
) ON COMMIT DROP;

INSERT INTO _varda_owner_tables (table_name) VALUES
  ('assets'),
  ('accounts'),
  ('asset_groups'),
  ('asset_group_members'),
  ('event_ledger_entries'),
  ('market_regime_daily'),
  ('goals'),
  ('transactions'),
  ('fixed_transactions'),
  ('monthly_incomes'),
  ('account_balance_snapshots'),
  ('daily_portfolio_snapshots'),
  ('daily_position_snapshots'),
  ('settings');

CREATE TEMP TABLE _varda_owner_counts_before (
  table_name text PRIMARY KEY,
  row_count bigint NOT NULL,
  null_owner_count bigint NOT NULL,
  requested_owner_count bigint NOT NULL,
  other_owner_count bigint NOT NULL
) ON COMMIT DROP;

CREATE TEMP TABLE _varda_backfill_state (
  status text PRIMARY KEY CHECK (status IN ('ready', 'already_assigned'))
) ON COMMIT DROP;

DO $$
DECLARE
  initial_owner uuid;
  writer_safety text;
  owned_table record;
  row_count bigint;
  null_count bigint;
  requested_count bigint;
  other_count bigint;
  total_rows bigint := 0;
  total_null bigint := 0;
  total_requested bigint := 0;
  total_other bigint := 0;
  canonical_column_count integer;
  legacy_owner_distinct_count integer;
BEGIN
  BEGIN
    initial_owner := current_setting('varda.initial_owner_id', true)::uuid;
  EXCEPTION WHEN OTHERS THEN
    RAISE EXCEPTION 'owner_backfill_invalid_initial_owner';
  END;

  writer_safety := current_setting('varda.writer_safety', true);
  IF writer_safety IS NULL
     OR writer_safety !~ '^(freeze|dual-write):[A-Za-z0-9._-]+$' THEN
    RAISE EXCEPTION 'owner_backfill_writer_safety_not_confirmed';
  END IF;

  IF NOT EXISTS (
    SELECT 1
    FROM app_users
    WHERE id = initial_owner
      AND status IN ('provisioning', 'active')
  ) THEN
    RAISE EXCEPTION 'owner_backfill_initial_app_user_not_eligible';
  END IF;

  SELECT count(*)::int
  INTO canonical_column_count
  FROM information_schema.columns c
  INNER JOIN _varda_owner_tables t ON t.table_name = c.table_name
  WHERE c.table_schema = 'public'
    AND c.column_name = 'canonical_owner_user_id'
    AND c.data_type = 'uuid'
    AND c.is_nullable = 'YES';

  IF canonical_column_count <> 14 THEN
    RAISE EXCEPTION 'owner_backfill_whitelist_schema_mismatch';
  END IF;

  IF EXISTS (
    SELECT 1
    FROM information_schema.columns c
    LEFT JOIN _varda_owner_tables t ON t.table_name = c.table_name
    WHERE c.table_schema = 'public'
      AND c.column_name = 'canonical_owner_user_id'
      AND t.table_name IS NULL
  ) THEN
    RAISE EXCEPTION 'owner_backfill_unapproved_canonical_owner_column';
  END IF;

  SELECT count(DISTINCT legacy_owner)::int
  INTO legacy_owner_distinct_count
  FROM (
    SELECT owner_user_id AS legacy_owner FROM accounts
    UNION ALL SELECT owner_user_id FROM asset_groups
    UNION ALL SELECT owner_user_id FROM asset_group_members
    UNION ALL SELECT owner_user_id FROM goals
    UNION ALL SELECT owner_user_id FROM transactions
    UNION ALL SELECT owner_user_id FROM fixed_transactions
    UNION ALL SELECT owner_user_id FROM monthly_incomes
  ) legacy
  WHERE legacy_owner IS NOT NULL;

  IF legacy_owner_distinct_count > 1 THEN
    RAISE EXCEPTION 'owner_backfill_legacy_owner_domain_changed';
  END IF;

  FOR owned_table IN
    SELECT table_name FROM _varda_owner_tables ORDER BY table_name
  LOOP
    EXECUTE format(
      'SELECT count(*), '
      'count(*) FILTER (WHERE canonical_owner_user_id IS NULL), '
      'count(*) FILTER (WHERE canonical_owner_user_id = $1), '
      'count(*) FILTER (WHERE canonical_owner_user_id IS NOT NULL '
      'AND canonical_owner_user_id <> $1) FROM %I',
      owned_table.table_name
    )
    INTO row_count, null_count, requested_count, other_count
    USING initial_owner;

    INSERT INTO _varda_owner_counts_before VALUES (
      owned_table.table_name,
      row_count,
      null_count,
      requested_count,
      other_count
    );

    total_rows := total_rows + row_count;
    total_null := total_null + null_count;
    total_requested := total_requested + requested_count;
    total_other := total_other + other_count;
  END LOOP;

  IF total_other > 0 THEN
    RAISE EXCEPTION 'owner_backfill_other_canonical_owner_present';
  END IF;

  IF total_null > 0 AND total_requested > 0 THEN
    RAISE EXCEPTION 'owner_backfill_partial_assignment_requires_investigation';
  END IF;

  IF total_rows = total_requested AND total_null = 0 THEN
    INSERT INTO _varda_backfill_state VALUES ('already_assigned');
  ELSE
    INSERT INTO _varda_backfill_state VALUES ('ready');
  END IF;
END $$;

DO $$
DECLARE
  initial_owner uuid := current_setting('varda.initial_owner_id', true)::uuid;
  owned_table record;
BEGIN
  IF EXISTS (SELECT 1 FROM _varda_backfill_state WHERE status = 'ready') THEN
    FOR owned_table IN
      SELECT table_name FROM _varda_owner_tables ORDER BY table_name
    LOOP
      EXECUTE format(
        'UPDATE %I SET canonical_owner_user_id = $1 '
        'WHERE canonical_owner_user_id IS NULL',
        owned_table.table_name
      ) USING initial_owner;
    END LOOP;
  END IF;
END $$;

DO $$
DECLARE
  initial_owner uuid := current_setting('varda.initial_owner_id', true)::uuid;
  owned_table record;
  before_count bigint;
  after_count bigint;
  invalid_owner_count bigint;
BEGIN
  FOR owned_table IN
    SELECT table_name FROM _varda_owner_tables ORDER BY table_name
  LOOP
    SELECT row_count INTO before_count
    FROM _varda_owner_counts_before
    WHERE table_name = owned_table.table_name;

    EXECUTE format(
      'SELECT count(*), count(*) FILTER '
      '(WHERE canonical_owner_user_id IS DISTINCT FROM $1) FROM %I',
      owned_table.table_name
    ) INTO after_count, invalid_owner_count USING initial_owner;

    IF after_count <> before_count THEN
      RAISE EXCEPTION 'owner_backfill_row_count_changed';
    END IF;
    IF invalid_owner_count <> 0 THEN
      RAISE EXCEPTION 'owner_backfill_owner_assignment_incomplete';
    END IF;
  END LOOP;
END $$;

-- Existence and same-owner relationship validation. Nullable resource ids are
-- intentionally skipped by each inner join.
DO $$
BEGIN
  IF EXISTS (
    SELECT 1 FROM assets c JOIN accounts p ON p.id = c.account_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM assets c JOIN asset_groups p ON p.id = c.group_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM asset_group_members c JOIN asset_groups p ON p.id = c.group_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM asset_group_members c JOIN assets p ON p.id = c.asset_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM event_ledger_entries c JOIN accounts p ON p.id = c.account_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM event_ledger_entries c JOIN assets p ON p.id = c.asset_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM event_ledger_entries c JOIN asset_groups p ON p.id = c.group_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM event_ledger_entries c JOIN event_ledger_entries p
      ON p.id = c.corrects_event_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM market_regime_daily c JOIN accounts p ON p.id = c.account_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM transactions c JOIN accounts p ON p.id = c.account_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM daily_portfolio_snapshots c JOIN accounts p ON p.id = c.account_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM daily_position_snapshots c JOIN accounts p ON p.id = c.account_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
    UNION ALL
    SELECT 1 FROM daily_position_snapshots c JOIN assets p ON p.id = c.asset_id
    WHERE c.canonical_owner_user_id IS DISTINCT FROM p.canonical_owner_user_id
  ) THEN
    RAISE EXCEPTION 'owner_backfill_same_owner_relationship_mismatch';
  END IF;
END $$;

-- Approved tenant-aware uniqueness candidates. Regime and balance snapshots
-- are deliberately absent.
DO $$
BEGIN
  IF EXISTS (
    SELECT 1 FROM accounts
    GROUP BY canonical_owner_user_id, code HAVING count(*) > 1
    UNION ALL
    SELECT 1 FROM asset_groups
    GROUP BY canonical_owner_user_id, name HAVING count(*) > 1
    UNION ALL
    SELECT 1 FROM asset_group_members
    GROUP BY canonical_owner_user_id, group_id, asset_id HAVING count(*) > 1
    UNION ALL
    SELECT 1 FROM monthly_incomes
    GROUP BY canonical_owner_user_id, year, month HAVING count(*) > 1
    UNION ALL
    SELECT 1 FROM daily_portfolio_snapshots
    GROUP BY canonical_owner_user_id, snapshot_date, account, source
    HAVING count(*) > 1
    UNION ALL
    SELECT 1 FROM daily_position_snapshots
    WHERE asset_id IS NOT NULL
    GROUP BY canonical_owner_user_id, snapshot_date, account, asset_id, source
    HAVING count(*) > 1
    UNION ALL
    SELECT 1 FROM settings
    GROUP BY canonical_owner_user_id HAVING count(*) > 1
  ) THEN
    RAISE EXCEPTION 'owner_backfill_tenant_unique_candidate_collision';
  END IF;
END $$;

-- The future runner statically records that its DML target set is exactly the
-- 14-table whitelist and contains zero shared/admin tables. Shared/admin row
-- counts are allowed to change through their independent writers.

SELECT status FROM _varda_backfill_state;

COMMIT;
